\def\DWOP#1{DW\_OP\_#1}

\def\stackdiagramprologue{
  \begin{tikzpicture}[
    every node/.style={
      node font=\small, align=center, anchor=north west,
      text height=.8em, text depth=.2em, text width=1.25in,
      inner sep=4pt, outer ysep=0, minimum height=14pt},
    stackelement/.style={draw},
    stackbottom/.style={
      append after command={
        \pgfextra
          \draw (\tikzlastnode.south west) --
                (\tikzlastnode.north west) --
                (\tikzlastnode.north east) --
                (\tikzlastnode.south east);
        \endpgfextra}}
  ]
}
\def\stackdiagramepilogue{\end{tikzpicture}}
\def\stackdiagramstart#1{
  \if#1L\def\XSHIFT{0}\else\def\XSHIFT{2in}\fi
  \begin{scope}[name prefix=#1, start chain=going above, node distance=0, xshift=\XSHIFT]}
\def\stackdiagramend{\end{scope}}
\def\drawstackbottom{\node (N0) [on chain, stackbottom] {\dots};}
\def\drawstackelement#1#2{\node (N#1) [on chain, stackelement] {#2};}
\def\drawstackarrow#1-#2.{\draw [-Stealth] (LN#1.east) -- (RN#2.west);}
\def\drawstackarrows#1{\foreach \A in {#1} {\expandafter\drawstackarrow \A.}}
\def\stackdiagram#1#2#3{
  \begin{center}
  \stackdiagramprologue
  \stackdiagramstart L
  \drawstackbottom
  \foreach \E [count=\N] in {#1} {\drawstackelement{\N}{\E}}
  \stackdiagramend
  \stackdiagramstart R
  \drawstackbottom
  \foreach \E [count=\N] in {#2} {\drawstackelement{\N}{\E}}
  \stackdiagramend
  %\foreach \A in {#3} {\show\A \drawstackarrow \A}
  \drawstackarrows {#3}
  \stackdiagramepilogue
  \end{center}
}
